.Data
	const TRUE 1
	const LEFT 1
	const RIGHT 0

	INPUT[2]
.Code
procstart main
start: 
	mov ax, (INPUT[0])
	mov bx, (INPUT[1])
	int 1
	mov ax, INPUT[0]
	mov bx, INPUT[1]
	cmp ax, TRUE
	jeq left_start
	cmp bx, TRUE
	jeq right_start
stop: 
	mov ax, 0
	mov bx, 0
	int 0
	jmp start
left_start: 
	mov ax, LEFT
	push ax
	call move
	jmp start
right_start: 
	mov ax, RIGHT
	push ax
	call move
	jmp start
procend main

procstart move
	pop ax
	mov bx, 10000
	int 0
	ret
procend mov
